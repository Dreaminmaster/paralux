function shiftPerspective(eye) {
  // screen is a fixed window
  var hw = screenW / 2;
  var d = eyeDistance;
 
  left = -(hw + eye.x)
    * (near / d);
  right = (hw - eye.x)
    * (near / d);
 
  projMatrix.makePerspective(
    left, right, top, bottom
  );
}
